# upgrade packages
sudo apt update
sudo apt upgrade

# disable serial input/output interface
sudo raspi-config
--> disable login shell over serial
--> enable serial port hardware

# install adafruit gps python libary
pip3 install adafruit-circuitpython-gps

# get internet on pi
sudo iwlist wlan0 scan
sudo vim /etc/wpa_supplicant/wpa_supplicant.conf:

    ctrl_interface=DIR=/var/run/wpa_supplicant GROUP=netdev
    update_config=1
    country=CA

    network={
        ssid="You SSID Name"
        psk="Your WiFI Password"
        key_mgmt=WPA-PSK
    }

note: can view wpa_supplicant log messages using
    * sudo wpa_supplicant -c/etc/wpa_supplicant/wpa_supplicant.conf -iwlan0 -d

# install PySimpleGUI
pip3 install PySimpleGUI

    --> note for mac (need to configure python 3 for tk): brew install python3-tk@3.9
